Begin 
     number num=1,even=0,odd=0
     while num<=10
          if num%2==0
             {
             even=even+num
             num=num+1
             }  
             else{
             odd=odd+num
             num=num+1
             }
     print("sum of even numbers of first 10 numbers is :",even)
     print("sum of odd numbers of first 10 numbers is :",odd)
End
